% packages.tex – Word-kompatible Formatierung (TNR 12pt)
% Kompiler: XeLaTeX
%
% Ziel: Maximale Annäherung an Word mit klassischer TNR-Konfiguration.
% Unvermeidliche Abweichungen sind am Ende dieser Datei dokumentiert.

% ──────────────────────────────────────────────
% Sprache & Schrift
% ──────────────────────────────────────────────
% babel: breiteste Paketkompatibilität, aktiv gepflegt, funktioniert
% einwandfrei mit XeLaTeX. ngerman = neue Rechtschreibung.
% Silbentrennung ist AKTIV (bewusste Abweichung von Word, das sie
% standardmäßig deaktiviert hat).
\usepackage[ngerman]{babel}

\usepackage{fontspec}
\setmainfont{Times New Roman}

% Mikrotypographie – Randausgleich, Ligaturen etc.
\usepackage{microtype}
\hyphenpenalty=500
\exhyphenpenalty=50
\tolerance=400

% Witwen-/Waisenzeilen verhindern
\widowpenalty=10000
\clubpenalty=10000

% ──────────────────────────────────────────────
% Zitate & Literatur
% ──────────────────────────────────────────────
\usepackage{csquotes}           % vor biblatex laden
\usepackage[
  backend=biber,
  style=authoryear                % Chicago
]{biblatex}
\addbibresource{/home/marcel/references.bib}

% Für Kurzzitate, wie z.B. Wittgenstein oder Aristoteles
\DeclareCiteCommand{\shortcite}[\mkbibparens]
  {\usebibmacro{prenote}}
  {\printfield{shorttitle}}
  {\multicitedelim}
  {\addspace\printfield{postnote}}
\DeclareFieldFormat{postnote}{#1} % entfernt p. vor Seitenzahl

% ──────────────────────────────────────────────
% Seitenlayout – Word „Normal" = 2,54 cm rundum
% ──────────────────────────────────────────────
% footskip: Word-Footer bei 1,25 cm vom Blattrand
%   → footskip = bottom − footer_pos = 2,54 − 1,25 = 1,29 cm
\usepackage[
  a4paper,
  top=2.54cm,
  bottom=2.54cm,
  left=2.54cm,
  right=2.54cm,
  headheight=14.5pt,
  footskip=1.29cm
]{geometry}

% ──────────────────────────────────────────────
% Zeilenabstand
% ──────────────────────────────────────────────
% Word „1,5 Zeilen" bei TNR 12pt = 12 × 1,2 × 1,5 = 21,6pt
% LaTeX: \baselineskip 14,5pt × 1,5 = 21,75pt (Δ +0,15pt ≈ 0,05mm)
% ACHTUNG: \onehalfspacing wäre nur Faktor 1,25 → ~18pt – NICHT
% äquivalent zu Word und deshalb hier explizit NICHT verwendet!
\usepackage{setspace}
\setstretch{1.5}

% ──────────────────────────────────────────────
% Absatzformatierung
% ──────────────────────────────────────────────
\setlength{\parindent}{0pt}
\setlength{\parskip}{8pt plus 0pt minus 0pt}   % Word-Default seit 2007

% ──────────────────────────────────────────────
% Kopf- und Fußzeile
% ──────────────────────────────────────────────
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}

% ──────────────────────────────────────────────
% Überschriften – exakte Word-TNR-Konfiguration
% ──────────────────────────────────────────────
% \fontsize{X}{Y}: X = Schriftgröße, Y = Baseline-Abstand (= X × 1,2)
% \linespread{1.0}\selectfont setzt den globalen Stretch (1,5) innerhalb
% der Überschrift auf 1,0 zurück, damit mehrzeilige Überschriften den
% richtigen Zeilenabstand haben. Word behandelt Headings ebenso.
\usepackage{titlesec}

% h1 = \section: 16pt fett, 12pt davor, 6pt danach
\titleformat{\section}
  {\normalfont\fontsize{16}{19.2}\linespread{1.0}\selectfont\bfseries}
  {\thesection.}{1em}{}
\titlespacing*{\section}{0pt}{12pt}{6pt}

% h2 = \subsection: 14pt fett, 10pt davor, 6pt danach
\titleformat{\subsection}
  {\normalfont\fontsize{14}{16.8}\linespread{1.0}\selectfont\bfseries}
  {\thesubsection.}{1em}{}
\titlespacing*{\subsection}{0pt}{10pt}{6pt}

% h3 = \subsubsection: 13pt fett, 8pt davor, 4pt danach
\titleformat{\subsubsection}
  {\normalfont\fontsize{13}{15.6}\linespread{1.0}\selectfont\bfseries}
  {\thesubsubsection.}{1em}{}
\titlespacing*{\subsubsection}{0pt}{8pt}{4pt}

% ──────────────────────────────────────────────
% Fußnoten: 10pt, einzeilig
% ──────────────────────────────────────────────
% \footnotesize ist in der 12pt-Klasse bereits 10pt.
% Einzeilig (\singlespacing) überschreibt den globalen 1,5-Stretch.
% Trennlinie: LaTeX-Default (~⅓ Spaltenbreite) ≈ Word (~5 cm).
\makeatletter
\renewcommand{\@makefntext}[1]{%
  \noindent\hb@xt@1.8em{\hss\@makefnmark}%
  {\singlespacing #1}}
\makeatother

% ──────────────────────────────────────────────
% Inhaltsverzeichnis (tocloft)
% ──────────────────────────────────────────────
\usepackage{tocloft}

% Überschrift „Inhaltsverzeichnis" wie \section (16pt fett)
\renewcommand{\cfttoctitlefont}{\fontsize{16}{19.2}\selectfont\bfseries}
\renewcommand{\cftaftertoctitle}{\par\vspace{6pt}}

% Schrift wie im Dokument (nicht fett)
\renewcommand{\cftsecfont}{\normalfont}
\renewcommand{\cftsubsecfont}{\normalfont}
\renewcommand{\cftsubsubsecfont}{\normalfont}

% Seitenzahlen nicht fett
\renewcommand{\cftsecpagefont}{\normalfont}
\renewcommand{\cftsubsecpagefont}{\normalfont}
\renewcommand{\cftsubsubsecpagefont}{\normalfont}

% Punktlinien zwischen Titel und Seitenzahl (Word-Stil)
\renewcommand{\cftdotsep}{1}
\renewcommand{\cftsecleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftsubsecleader}{\cftdotfill{\cftdotsep}}
\renewcommand{\cftsubsubsecleader}{\cftdotfill{\cftdotsep}}

% Abstände zwischen Einträgen
\setlength{\cftbeforesecskip}{2pt}
\setlength{\cftbeforesubsecskip}{2pt}
\setlength{\cftbeforesubsubsecskip}{2pt}

% Einrückungen (Hierarchie)
\setlength{\cftsecindent}{0pt}
\setlength{\cftsubsecindent}{1.5em}
\setlength{\cftsubsubsecindent}{3em}

% Nummernbreiten
\setlength{\cftsecnumwidth}{1.8em}
\setlength{\cftsubsecnumwidth}{2.6em}
\setlength{\cftsubsubsecnumwidth}{3.4em}

% Punkt nach der Nummer
\renewcommand{\cftsecaftersnum}{.}
\renewcommand{\cftsubsecaftersnum}{.}
\renewcommand{\cftsubsubsecaftersnum}{.}

% ──────────────────────────────────────────────
% Listen
% ──────────────────────────────────────────────
\usepackage{enumitem}
\setlist{itemsep=4pt, parsep=0pt, topsep=4pt}

% ──────────────────────────────────────────────
% Hyperlinks – alles schwarz, keine Rahmen
% ──────────────────────────────────────────────
\usepackage[hidelinks]{hyperref}
\hypersetup{colorlinks=false, allcolors=black}

% ──────────────────────────────────────────────
% Blockzitate – nur links eingerückt (dt. wiss. Standard)
% ──────────────────────────────────────────────
% 1,27 cm = 0,5 Zoll = Word-Standard-Einrückung
\renewenvironment{displayquote}
  {\list{}{\leftmargin=1.27cm \rightmargin=0cm}\item\relax}
  {\endlist}

% ──────────────────────────────────────────────
% BEKANNTE ABWEICHUNGEN VON WORD
% ──────────────────────────────────────────────
% 1. Zeilenumbrüche: Word nutzt Greedy-Algorithmus, LaTeX Knuth-Plass
%    (optimal) → unterschiedliche Umbruchstellen bei langen Absätzen.
% 2. Baseline-Abstand: Word 21,6pt vs. LaTeX 21,75pt (Δ +0,15pt).
% 3. Silbentrennung: Hier aktiv, bei Word standardmäßig aus.
% 4. Schriftmetriken/Hinting: Minimale Laufweitenunterschiede durch
%    unterschiedliche Rendering-Engines.
% 5. Fußnotentrennlinie: LaTeX ~⅓ Spaltenbreite vs. Word ~5 cm.
